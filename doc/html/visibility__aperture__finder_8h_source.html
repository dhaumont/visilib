75  VisibilityResult VisibilityApertureFinder<P, S>::resolve()
76  {
77  VisibilityResult result = resolveInternal(reinterpret_cast<PluckerPolytope<P>*>(VisibilitySolver<P, S>::mQuery->getComplex()->getFirstPolytope()), 0);
78  if (result == UNKNOWN)
79  {
80  // The source polygons are not mutually visible: no aperture has been found, this means that all the polytopes have been deleted and all the stabbing lines are blocked by occluders.
81  result = HIDDEN;
82  }
83  return result;

86  template<class P, class S>
87  VisibilityResult VisibilityApertureFinder<P, S>::resolveInternal(PluckerPolytope<P>* aPolytope, int aDeep)
88  {
89  PluckerPolyhedron<P>* myPolyhedron = reinterpret_cast<PluckerPolyhedron<P>*> (VisibilitySolver<P, S>::mQuery->getComplex()->getPolyhedron());
90 
91  if (aDeep > 500)
92  {
93  std::cerr << "Overflow detected: more than 200 recursive calls...." << std::endl;
94  V_LOG("overflow", aDeep);
95 
96  return FAILURE;
97  }
98  {
99  HelperScopedTimer timer(VisibilitySolver<P, S>::mQuery->getStatistic(), STABBING_LINE_EXTRACTION);
100  aPolytope->computeEdgesIntersectingQuadric(myPolyhedron, mTolerance);
101  }
102  if (!aPolytope->isValid())
103  {
104  V_LOG("isValid = false", aDeep);
105  return UNKNOWN;
106  }
107 
108  V_ASSERT(aPolytope->isValid(myPolyhedron, mNormalization, mTolerance));
109 
110  if (!VisibilitySolver<P, S>::mQuery->collectAllOccluders(aPolytope, myPolyhedron))
111  {
112  // Early stop - an aperture has been found: the source polygons are mutually visible
113 
114  V_LOG("collect all occluder = false", aDeep);
115 
116  return VISIBLE;
117  }
118 
119  VisibilitySilhouette* mySilhouette = nullptr;
120  GeometryMeshFace* face = nullptr;
121  size_t edgeIndex = 0;
122 
123  bool hasEdge = false;
124  {
125  // We look for a valid candidate edge in the occluder set
126  //HelperScopedTimer timerVisibilitySolver<P, S>::(VisibilitySolver<P, S>::mQuery->getStatistic(), OCCLUDER_SELECTION);
127  hasEdge = VisibilitySolver<P, S>::mQuery->findTheBestValidEdge(face, edgeIndex, mySilhouette, aPolytope);
128  }
129  if (hasEdge) // a candidate edge has been found. we will split the polytope with the hyperplane of this edge.
130  {
131  // deactivate the candidate edge to avoid infinite recusion loop
132  mySilhouette->remove(face, edgeIndex);
133 
134  const MathVector2i& edge = face->getEdge(edgeIndex);
135  if (VisibilitySolver<P, S>::mDebugger != nullptr)
136  {
137  VisibilitySolver<P, S>::mDebugger->get()->addRemovedEdge(face->getVertex(edge.x), face->getVertex(edge.y));
138  }
139 
140  // Create the Plucker representation of the edge and add hyperplane of the edge and add it to the polyhedron
141  P myHyperplane(convert<MathVector3d>(face->getVertex(edge.x)), convert<MathVector3d>(face->getVertex(edge.y)));
142  if (mNormalization)
143  {
144  myHyperplane = myHyperplane.getNormalized();
145  }
146 
147  size_t myPolyhedronFace = myPolyhedron->add(myHyperplane, ON_BOUNDARY, mNormalization, mTolerance);
148 
149  PluckerPolytope<P> myPolytopeLeft;
150  PluckerPolytope<P> myPolytopeRight;
151 
152  // Try to split the polytope by the hyperplane of the candidate edge
153 
154  GeometryPositionType myResult = ON_BOUNDARY;
155  {
156  HelperScopedTimer timer(VisibilitySolver<P, S>::mQuery->getStatistic(), POLYTOPE_SPLIT);
157  VisibilitySolver<P, S>::mQuery->getStatistic()->inc(POLYTOPE_SPLIT_COUNT);
158  myResult = PluckerPolytopeSplitter<P, S>::split(myPolyhedron, myHyperplane, aPolytope, &myPolytopeLeft, &myPolytopeRight, myPolyhedronFace, mNormalization, mTolerance);
159  }
160 
161  if (myResult == ON_BOUNDARY)
162  {
163  // The polytope has been split in two polytopes - we continue to recurse on both polytopes
164 
165  V_LOG("resut = intersect", aDeep);
166 
167  VisibilityResult resultLeft = resolveInternal(&myPolytopeLeft, aDeep + 1);
168  if (resultLeft != UNKNOWN) // Early stop - an aperture has been found
169  return resultLeft;
170 
171  VisibilityResult resultRight = resolveInternal(&myPolytopeRight, aDeep + 1);
172  if (resultRight != UNKNOWN) // Early stop - an aperture has been found
173  return resultRight;
174 
175  VisibilitySolver<P, S>::mQuery->unlink(mySilhouette, &myPolytopeLeft);
176  VisibilitySolver<P, S>::mQuery->unlink(mySilhouette, &myPolytopeRight);
177  }
178  else
179  {
180  // The polytope was no split - we continue to recurse (with the candidate silhouette edge deactivated)
181 
182  V_LOG("resut is not intersect", aDeep);
183 
184  VisibilityResult r = resolveInternal(aPolytope, aDeep + 1);
185  if (r != UNKNOWN)
186  {
187  V_LOG("resolve internal = true", aDeep);
188  return r;
189  }
190  }
191 
192  // reactivate the candidate edge
193  mySilhouette->add(face, edgeIndex);
194  }
195 
196  // No valid candidate edge has been found in the occluder set: all the set of lines represented by the polytope are blocked by at least one occluder: the recursion can stops.
197  // When it is the case for all the sub-polytopes created during the recursive traversal, the two polygons are mutually hidden.
198  return UNKNOWN;
199  }
