151  template<typename A, typename B>
152  static void link(std::unordered_map<A, std::unordered_set<B>>& map, A a, B b)
153  {
154  auto iter = map.find(a);
155  if (iter == map.end())
156  {
157  map.insert(std::pair<A, std::unordered_set<B>>(a, std::unordered_set<B>()));
158  iter = map.find(a);
159  }
160 
161  if (iter->second.find(b) == iter->second.end())
162  {
163  iter->second.insert(b);
164  }
165  }

170  template<typename A, typename B>
171  static void unlink(std::unordered_map<A, std::unordered_set<B>>& map, A a, B b)
172  {
173  auto iter = map.find(a);
174  if (iter != map.end())
175  {
176  if (iter->second.find(b) != iter->second.end())
177  {
178  iter->second.erase(b);
179  }
180  }

264  template<class P, class S>
265  bool VisibilityExactQuery<P, S>::createInitialPolygons(const float* vertices0, size_t numVertices0, const float* vertices1, size_t numVertices1, bool)
266  {
267  GeometryConvexPolygon myQuery0(vertices0, numVertices0);
268  GeometryConvexPolygon myQuery1(vertices1, numVertices1);
269 
270  if (myQuery0.getVertexCount() < 3 || myQuery1.getVertexCount() < 3)
271  {
272  MathVector3d g1 = MathGeometry::getGravityCenter(myQuery0);
273  MathVector3d g2 = MathGeometry::getGravityCenter(myQuery1);
274 
275  MathVector3d myN = g2 - g1;
276 
277  if (myQuery0.getVertexCount() == 2)
278  {
279  myQuery0.setPlane(MathGeometry::computePlaneFromApproximateNormal(myQuery0, myN));
280  }
281  else if (myQuery0.getVertexCount() == 1)
282  {
283  MathVector3d myNormal = myN;
284  myNormal.normalize();
285  myQuery0.setPlane(MathPlane3d(myNormal.x, myNormal.y, myNormal.z, -myNormal.dot(myQuery0.getVertex(0))));
286  }
287 
288  if (myQuery1.getVertexCount() == 2)
289  {
290  myN.x = -myN.x; myN.y = -myN.y; myN.z = -myN.z;
291  myQuery1.setPlane(MathGeometry::computePlaneFromApproximateNormal(myQuery1, myN));
292  }
293  else if (myQuery1.getVertexCount() == 1)
294  {
295  myN.x = -myN.x; myN.y = -myN.y; myN.z = -myN.z;
296  MathVector3d myNormal = myN;
297  myNormal.normalize();
298  myQuery1.setPlane(MathPlane3d(myNormal.x, myNormal.y, myNormal.z, -myNormal.dot(myQuery1.getVertex(0))));
299  }
300  }
301 
302  if (!MathGeometry::clipWithGardBand(myQuery0, myQuery1.getPlane(), MathArithmetic<double>::GuardBandClipping()))
303  {
304  std::cout << "Clipping failure" << std::endl;
305  return false;
306  }
307  V_ASSERT(myQuery0.isValid());
308 
309  if (!MathGeometry::clipWithGardBand(myQuery1, myQuery0.getPlane(), MathArithmetic<double>::GuardBandClipping()))
310  {
311  std::cout << "Clipping failure" << std::endl;
312  return false;
313  }
314  V_ASSERT(myQuery1.isValid());
315 
316  V_ASSERT(MathPredicates::getRelativePosition(myQuery0.getVertices(), myQuery1.getPlane()) == ON_POSITIVE_SIDE);
317  V_ASSERT(MathPredicates::getRelativePosition(myQuery1.getVertices(), myQuery0.getPlane()) == ON_POSITIVE_SIDE);
318 
319  mQueryPolygon[0] = new GeometryConvexPolygon(myQuery0);
320  mQueryPolygon[1] = new GeometryConvexPolygon(myQuery1);
321 
322  return true;
323  }

325  template<class P, class S>
326  VisibilityResult VisibilityExactQuery<P, S>::arePolygonsVisible(const float* vertices0, size_t numVertices0, const float* vertices1, size_t numVertices1)
327  {
328  VisibilityResult result;
329 
330  HelperScopedTimer timer(&mStatistic, VISIBILITY_QUERY);
331 
332  if (mDebugger != nullptr)
333  {
334  mDebugger->get()->clear();
335  mSilhouetteProcessor->attachVisualisationDebugger(mDebugger);
336  }
337 
338  if (!createInitialPolygons(vertices0, numVertices0, vertices1, numVertices1, mConfiguration.hyperSphereNormalization))
339  {
340  std::cout << "Polygon initialization error" << std::endl;
341  return FAILURE;
342  }
343 
344  if (mQueryPolygon[0]->getVertexCount() == 1 && mQueryPolygon[1]->getVertexCount() == 1)
345  {
346  result = findSceneIntersection(mQueryPolygon[0]->getVertex(0), mQueryPolygon[1]->getVertex(0)) ? HIDDEN : VISIBLE;
347  }
348  else
349  {
350  //TODO remove this and better manage clipping
351  mScene->get()->restoreFacesGeometry(mScene);
352  mSilhouetteProcessor->init(*mQueryPolygon[0], *mQueryPolygon[1]);
353 
354  {
355  HelperScopedTimer timerBuild(&mStatistic, POLYTOPE_BUILD);
356 
357  PluckerPolytopeBuilder<P, S> builder(mConfiguration.hyperSphereNormalization, mTolerance);
358  PluckerPolytope<P>* myPolytope = builder.build(*mQueryPolygon[0], *mQueryPolygon[1], getComplex()->getPolyhedron());
359  getComplex()->setRoot(myPolytope);
360  }
361  VisibilitySolver<P, S>* solver;
362 
363  solver = new VisibilityApertureFinder<P, S>(this, mConfiguration.hyperSphereNormalization, mTolerance, mConfiguration.performFastTest);
364 
365  if (mDebugger != nullptr)
366  {
367  solver->attachVisualisationDebugger(mDebugger);
368  }
369  result = solver->resolve();
370 
371  delete solver;
372  }
373 
374  return result;
375  }

383  template<class P, class S>
384  bool VisibilityExactQuery<P, S>::findTheBestValidEdge(GeometryMeshFace * &aFace, size_t & anEdgeIndex, VisibilitySilhouette * &aSilhouette, PluckerPolytope<P> * polytope)
385  {
386  V_LOG("findTheBestValidEdge");
387  if (!hasSilhouette(polytope))
388  {
389  V_LOG("occluded: no silhouette", 1);
390  return false;
391  }
392 
393  const std::unordered_set<VisibilitySilhouette*>& mySilhouettes = getSilhouettes(polytope);
394 
395  if (mConfiguration.silhouetteOptimization && mConfiguration.performFastTest)
396  {
397  for (auto iter = mySilhouettes.begin(); iter != mySilhouettes.end(); iter++)
398  {
399  VisibilitySilhouette* s = (*iter);
400  if (s->getEdgeCount() == 0)
401  {
402  V_LOG("occluded: silhouette without edge", 1);
403  return false;//occluded
404  }
405  }
406  }
407 
408  size_t myBestNumber = 0;//number of polytope for this silhouette
409  bool found = false;
410  GeometryMeshFace* myFace = nullptr;
411  size_t myEdgeIndex = 0;
412  VisibilitySilhouette* mySilhouette = nullptr;
413 
414  for (auto iter = mySilhouettes.begin(); iter != mySilhouettes.end(); iter++)
415  {
416  VisibilitySilhouette* s = (*iter);
417  GeometryMeshFace* f = nullptr;
418  size_t e = 0;
419 
420  if (s->findEdge(f, e))
421  {
422  size_t polytopeCount = getPolytopeCount(s);
423  if (polytopeCount >= myBestNumber)
424  {
425  found = true;
426  myBestNumber = polytopeCount;
427  mySilhouette = s;
428  myFace = f;
429  myEdgeIndex = e;
430  }
431  }
432  }
433 
434  if (found)
435  {
436  V_LOG("edge found", 1);
437  aFace = myFace;
438  anEdgeIndex = myEdgeIndex;
439  aSilhouette = mySilhouette;
440 
441  V_ASSERT(aSilhouette->hasEdge(aFace, anEdgeIndex));
442  }
443  else
444  {
445  V_LOG("occluded: edge not found", 1);
446  }
447  return found;
448  }

450  template<class P, class S>
451  bool VisibilityExactQuery<P, S>::findSceneIntersection(const MathVector3d & aBegin, const MathVector3d & anEnd, PluckerPolytope<P> * aPolytope)
452  {
453  MathVector3d myBegin = aBegin;
454  MathVector3d myDir = anEnd - aBegin;
455  double myMax = myDir.normalize();
456 
457 #if _DEBUG
458  V_ASSERT(myMax > MathArithmetic<float>::Tolerance());
459  for (int i = 0; i < 2; i++)
460  {
461  if (getQueryPolygon(i)->getVertexCount() > 1)
462  {
463  MathVector3d anIntersection;
464  V_ASSERT(MathGeometry::getPlaneIntersection(getQueryPolygon(i)->getPlane(), myBegin, myDir, anIntersection, MathArithmetic<float>::Tolerance()));
465  V_ASSERT(MathGeometry::isPointInsidePolygon(*getQueryPolygon(i), anIntersection, MathArithmetic<float>::Tolerance()));
466  }
467  }
468 #endif
469  Ray myRay;
470  myRay.org[0] = (float)myBegin.x;
471  myRay.org[1] = (float)myBegin.y;
472  myRay.org[2] = (float)myBegin.z;
473  myRay.dir[0] = (float)myDir.x;
474  myRay.dir[1] = (float)myDir.y;
475  myRay.dir[2] = (float)myDir.z;
476  myRay.tfar = (float)myMax;
477  myRay.tnear = 0;
478  myRay.primitiveId = 0;
479  myRay.geometryId = 0;
480 
481  if (mDebugger != nullptr)
482  {
483  mDebugger->get()->addSamplingLine(convert<MathVector3f>(aBegin), convert<MathVector3f>(anEnd));
484  }
485 
486  bool intersect = false;
487 
488  {
489  HelperScopedTimer timer(getStatistic(), RAY_INTERSECTION);
490  getStatistic()->inc(RAY_COUNT);
491  intersect = mScene->intersect(&myRay);
492  }
493 
494  if (intersect)
495  {
496  size_t myPrimitive = myRay.primitiveId;
497  size_t geometryId = myRay.geometryId;
498 
499  {
500  HelperScopedTimer timer(getStatistic(), SILHOUETTE_PROCESSING);
501 
502  std::vector<GeometryMeshFace>* myFaces = mScene->get()->getFaces(mScene, geometryId);
503 
504  GeometryMeshFace* myFace = &(*myFaces)[myPrimitive];
505 
506  VisibilitySilhouette* s = mSilhouetteProcessor->extractSilhouette(myFace, mConfiguration.silhouetteOptimization);
507 
508  link(s, aPolytope);
509  }
510  return true;
511  }
512 
513  if (mDebugger != nullptr)
514  {
515  mDebugger->get()->addStabbingLine(convert<MathVector3f>(aBegin), convert<MathVector3f>(anEnd));
516  }
517 
518  return false;
519  }

521  template<class P, class S>
522  bool VisibilityExactQuery<P, S>::collectAllOccluders(PluckerPolytope<P> * aPolytope, PluckerPolyhedron<P> * polyhedron)
523  {
524  std::vector<std::pair<MathVector3d, MathVector3d>> lines;
525 
526  {
527  HelperScopedTimer timer(getStatistic(), STABBING_LINE_EXTRACTION);
528 
529  MathPlane3d aPlane0 = getQueryPolygon(0)->getPlane();
530  MathPlane3d aPlane1 = getQueryPolygon(1)->getPlane();
531 
532  if (mConfiguration.representativeLineSampling)
533  {
534  std::pair<MathVector3d, MathVector3d> myStriking = MathGeometry::computeRepresentativeLine<P>(aPolytope, polyhedron, aPlane0, aPlane1);
535 
536  lines.push_back(myStriking);
537  }
538  else
539  {
540  if (aPolytope->getExtremalStabbingLinesCount() == 0)
541  {
542  aPolytope->computeExtremalStabbingLines(polyhedron, mTolerance);
543  }
544  aPolytope->getExtremalStabbingLinesBackTo3D(lines, aPlane0, aPlane1);
545  }
546  }
547  V_ASSERT(!lines.empty());
548 
549  for (size_t i = 0; i < lines.size(); i++)
550  {
551  bool hit = findSceneIntersection(lines[i].first, lines[i].second, aPolytope);
552 
553  if (!hit)
554  {
555  return false;
556  }
557  }
558  return true;
559  }
