60  {

80  inline bool GeometryConvexHullBuilder::compute(const std::vector<MathVector3d>& inputVertices, std::vector<MathPlane3d>& outputFaces)
81  {
82  ch::convex_hull myHull;
83 
84  for (size_t i = 0; i < inputVertices.size(); i++)
85  {
86  myHull.addOneVertice(inputVertices[i].x, inputVertices[i].y, inputVertices[i].z);
87  }
88 
89  myHull.compute();
90 
91  std::vector<ch::vertex> vertices;
92 
93  if (!myHull.getVertices(vertices))
94  {
95  V_ASSERT(0);
96  return false;
97  }
98 
99  MathVector3d v[3];
100  MathPlane3d myPlane;
101 
102  outputFaces.clear();
103 
104  for (size_t vertex = 0; vertex < vertices.size(); vertex += 3)
105  {
106  for (size_t index = 0; index < 3; index++)
107  {
108  v[index] = MathVector3d(vertices[vertex + index].x, vertices[vertex + index].y, vertices[vertex + index].z);
109  }
110 
111  if (MathPlane3d::computeTrianglePlane(myPlane, v, MathArithmetic<double>::Tolerance()))
112  {
113  outputFaces.push_back(myPlane);
114  }
115  else
116  {
117  V_ASSERT(0);
118  return false;
119  }
120  }
121 
122  return true;
123  }
